Documentation

entrance address: https://wg-force3-backend.herokuapp.com/api

methods GET

Get user by ID

query string address: /user

request data: {id:"string"}
example: {id:"61a6286353b5dad92e57b4c0"}

response data: { resultCode: 0 | 1, messages: [string], data: { id: string, name: string, wishlist: [string], shoppingList: [string], } }

Get product by ID

query string address: /product

request data: {id:"string"}
example: {id:"61a61eb74ef1f443c0cb654e"}

response data:{ resultCode: 0 | 1, messages: [string], data: { type:string, span:string, data: productData } }

productData(technique):{ id: string, name: string, description: string, filter: { nation: string, type: string, tier: string }, price: { basic: { cost: string, currency: string, }, actual: { cost: string, currency: string, }, }, images: { span_1x1: string, span_2x1: string, }, }

productData(other type):{ id: string, name: string, description: string, price: { basic: { cost: string, currency: string, }, actual: { cost: string, currency: string, }, }, images: { span_1x1: string, span_2x1: string, }, }

Get products by filter

query string address: /product/filter

request data: {"filter":["string"]}
example:{"filter":["Technique"]}. Request ALL :{"filter":["Technique", "Premium", "Gold", "Provisions"]} }
some filter: "Technique", "Premium", "Gold", "Provisions"

response data:{ resultCode: 0 | 1, messages: [string], data: [{ type:string, span:string, data: productData }] }

productData(technique):{ id: string, name: string, description: string, filter: { nation: string, type: string, tier: string }, price: { basic: { cost: string, currency: string, }, actual: { cost: string, currency: string, }, }, images: { span_1x1: string, span_2x1: string, }, }

productData(other type):{ id: string, name: string, description: string, price: { basic: { cost: string, currency: string, }, actual: { cost: string, currency: string, }, }, images: { span_1x1: string, span_2x1: string, }, }

methods POST

Add user

query string address: /user

request data: {"name":"string"}
example:{"name":"Tester"}

response data: { resultCode: 0 | 1, messages: [string], data: { id: string, name: string, wishlist: [string], shoppingList: [string], } }

Change wishlist

query string address: /user/wishlist

request data: {"userId":"string", "productId":"string"}
example: {"userId":"61a6286353b5dad92e57b4c0", "productId":"61a61eb74ef1f443c0cb654e"}

response data: { resultCode: 0 | 1, messages: [string], data: { id: string, name: string, wishlist: [string], shoppingList: [string], } }

Change purchase

query string address: /user/purchase

request data: {"userId":"string","productId":"string" , isAdd:"boolean"}
example:{"userId":"61a6286353b5dad92e57b4c0", "productId":"61a61eb74ef1f443c0cb654e", isAdd:true}

response data: { resultCode: 0 | 1, messages: [string], data: { id: string, name: string, wishlist: [string], shoppingList: [string], } }
